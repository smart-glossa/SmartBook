create table User (
  userId int NOT NULL AUTO_INCREMENT PRIMARY KEY,
  name varchar(250),
  dob varchar(250),
  userName varchar(250),
  password varchar(250),
  imageId int,
  type varchar(5)
);

CREATE TABLE Project (
  projectId INT NOT NULL PRIMARY KEY,
  ptitle VARCHAR (250),
  dis VARCHAR(250),
  date varchar(100),
  status varchar(20)
)
